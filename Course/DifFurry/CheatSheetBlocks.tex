\[
\begin{aligned}
&f(x)=k &\Rightarrow& F(x)=kx\\
&f(x)=x^{n} &\Rightarrow& F(x)=\frac{x^{n+1}}{n+1}\\
&f(x)=\frac{1}{x} &\Rightarrow& F(x)=\ln|x|\\
&f(x)=a^{x} &\Rightarrow& F(x)=\frac{a^{x}}{\ln a}\\
&f(x)=\sin x &\Rightarrow& F(x)=-\cos x\\
&f(x)=\cos x &\Rightarrow& F(x)=\sin x\\
&f(x)=\csc^{2}x &\Rightarrow& F(x)=-\ctg x\\
&f(x)=\sec^{2}x &\Rightarrow& F(x)=\tg x\\
&f(x)=\frac{1}{\sqrt{1-x^{2}}} &\Rightarrow& F(x)=\arcsin x\\
&f(x)=\frac{1}{1+x^{2}} &\Rightarrow& F(x)=\arctg x
\end{aligned}
\]

% Derivatives
\[
\begin{aligned}
&(c)'&=&0\\
&(x^{\alpha})'&=&\alpha x^{\alpha-1}\\
&(a^{x})'&=&a^{x}\ln a,\,(e^x)'=e^x\\
&(\log_{a}x)'&=&\frac{1}{x\ln a},\,\ln x = \frac{1}{x}\\
&(\sin x)'&=&\cos x, (\cos x)'=-\sin x\\
&(\tg x)'&=&\sec^{2}x=\frac{1}{\cos^{2}x}\\
&(\ctg x)'&=&-\csc^{2}x=-\frac{1}{\sin^{2}x}\\
&(\arcsin x)'&=&\frac{1}{\sqrt{1-x^{2}}},\ (\arccos x)'=-\frac{1}{\sqrt{1-x^{2}}}\\
&(\arctg x)'&=&\frac{1}{1+x^{2}},\, (\arcctg x)'=-\frac{1}{1+x^{2}}.
\end{aligned}
\]

\[
\mathcal{L}\{f(t)\}(s)=\int_{0}^{\infty}e^{-st}f(t)\,dt=F(s)

\[
\begin{aligned}
&\mathcal{L}\{1\}&=&\frac{1}{s}\\
&\mathcal{L}\{t^{n}\}&=&\frac{n!}{s^{n+1}}\\
&\mathcal{L}\{\sqrt{t}\}&=&\frac{\sqrt{\pi}}{2\,s^{3/2}}\\
&\mathcal{L}\{\sin(at)\}&=&\frac{a}{s^{2}+a^{2}}\\
&\mathcal{L}\{t\sin(at)\}&=&\frac{2as}{(s^{2}+a^{2})^{2}}\\
&\mathcal{L}\{\cos(at)\}&=&\frac{s}{s^{2}+a^{2}}\\
&\mathcal{L}\{\sin(at+b)\}&=&\frac{s\sin b+a\cos b}{s^{2}+a^{2}}\\
&\mathcal{L}\{\sinh(at)\}&=&\frac{a}{s^{2}-a^{2}}\\
&\mathcal{L}\{e^{at}\sin(bt)\}&=&\frac{b}{(s-a)^{2}+b^{2}}\\
&\mathcal{L}\{e^{at}\sinh(bt)\}&=&\frac{b}{(s-a)^{2}-b^{2}}\\
&\mathcal{L}\{t^{n}e^{at}\}&=&\frac{n!}{(s-a)^{n+1}}\\
&\mathcal{L}\{u(t-c)\}&=&\frac{e^{-cs}}{s}\\
&\mathcal{L}\{e^{at}\}&=&\frac{1}{s-a}\\
&\mathcal{L}\{t^{p}\}&=&\frac{\Gamma(p+1)}{s^{p+1}},\quad p>-1\\
&\mathcal{L}\{t^{\,n-\tfrac12}\}&=&\frac{1\cdot3\cdot5\cdots(2n-1)\sqrt{\pi}}{2^{\,n}\,s^{\,n+\tfrac12}}\\
&\mathcal{L}\{t\cos(at)\}&=&\frac{s^{2}-a^{2}}{(s^{2}+a^{2})^{2}}\\
&\mathcal{L}\{\cos(at+b)\}&=&\frac{s\cos b-a\sin b}{s^{2}+a^{2}}\\
&\mathcal{L}\{\cosh(at)\}&=&\frac{s}{s^{2}-a^{2}}\\
&\mathcal{L}\{e^{at}\cos(bt)\}&=&\frac{s-a}{(s-a)^{2}+b^{2}}\\
&\mathcal{L}\{e^{at}\cosh(bt)\}&=&\frac{s-a}{(s-a)^{2}-b^{2}}\\
&\mathcal{L}\{f(ct)\}&=&\frac{1}{c}\,F\!\left(\frac{s}{c}\right)\quad(c>0)\\
&\mathcal{L}\{\delta(t-c)\}&=&e^{-cs}
\end{aligned}
\]
